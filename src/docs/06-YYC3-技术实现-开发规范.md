## 3. TypeScript 类型规范

### 3.1 类型定义原则

#### ✅ 优先使用 `interface`

```typescript
// ✅ 推荐：对象类型使用 interface
interface PanelSpec {
  id: string;
  type: 'container' | 'content' | 'preview';
  layout: {
    x: number;
    y: number;
    w: number;
    h: number;
  };
  style: PanelStyle;
}

// ❌ 避免：对象类型使用 type
type PanelSpec = {
  id: string;
  type: 'container' | 'content' | 'preview';
};
```

#### ✅ 联合类型使用 `type`

```typescript
// ✅ 推荐：联合类型使用 type
export type MessageRole = 'user' | 'ai' | 'system';
export type ViewMode = 'code' | 'preview' | 'fullscreen';
export type ThemeMode = 'light' | 'dark' | 'midnight' | 'forest' | 'sunset';

// ✅ 推荐：复杂组合类型使用 type
export type AIModelProvider = 'openai' | 'ollama' | 'custom';
```

  // ── 1. Zustand Store ──

  const updatePanel = useAppStore(state => state.updatePanel);

## 5. 状态管理规范

### 5.1 Zustand Store 设计

#### `/src/app/store.ts` - 全局状态

```typescript
/**
 * @file store.ts
 * @description YYC3 PortAISys - Global State Management
 * Zustand store with persist middleware for LocalStorage
 */

import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface AppState {
  // ── UI State ──
  theme: ThemeMode;
  setTheme: (theme: ThemeMode) => void;
  toggleTheme: () => void;

  viewMode: ViewMode;
  setViewMode: (mode: ViewMode) => void;

  // ── File State ──
  selectedFile: string | null;
  setSelectedFile: (file: string | null) => void;

  // ── Terminal State ──
  terminalVisible: boolean;
  toggleTerminal: () => void;
  terminalHeight: number;
  setTerminalHeight: (h: number) => void;

  // ── AI State ──
  messages: Message[];
  addMessage: (msg: Omit<Message, 'id' | 'timestamp'>) => void;
  clearMessages: () => void;

  aiModels: AIModel[];
  activeModelId: string | null;
  addAIModel: (model: Omit<AIModel, 'id'>) => void;
  removeAIModel: (id: string) => void;
  updateAIModel: (id: string, updates: Partial<AIModel>) => void;
  activateAIModel: (id: string) => void;

  // ── Panel State ──
  panelMap: { left: string; middle: string; right: string };
  setPanelMap: (map: { left: string; middle: string; right: string }) => void;
  swapPanels: (from: 'left' | 'middle' | 'right', to: 'left' | 'middle' | 'right') => void;

  // ── Collaboration State (yjs-driven) ──
  collaborators: Collaborator[];
  setCollaborators: (c: Collaborator[]) => void;

  // ── Code Injection (AI → CodeEditor) ──
  pendingCodeInjection: CodeInjection | null;
  injectCode: (filename: string, code: string, language: string) => void;
  clearCodeInjection: () => void;
}

export const useAppStore = create<AppState>()(
  persist(
    (set, get) => ({
      // ── Initial State ──
      theme: 'dark',
      viewMode: 'code',
      selectedFile: null,
      terminalVisible: false,
      terminalHeight: 200,
      messages: [],
      aiModels: [],
      activeModelId: null,
      panelMap: { left: 'chat', middle: 'file-manager', right: 'code-editor' },
      collaborators: [],
      pendingCodeInjection: null,

      // ── Actions ──
      setTheme: (theme) => set({ theme }),
      toggleTheme: () => set((state) => ({ theme: nextTheme(state.theme) })),

      setViewMode: (viewMode) => set({ viewMode }),

      setSelectedFile: (selectedFile) => set({ selectedFile }),

      toggleTerminal: () =>
        set((state) => ({
          terminalVisible: !state.terminalVisible,
        })),
      setTerminalHeight: (terminalHeight) => set({ terminalHeight }),

      addMessage: (msg) =>
        set((state) => ({
          messages: [
            ...state.messages,
            {
              ...msg,
              id: crypto.randomUUID(),
              timestamp: Date.now(),
            },
          ],
        })),
      clearMessages: () => set({ messages: [] }),

      addAIModel: (model) =>
        set((state) => ({
          aiModels: [...state.aiModels, { ...model, id: crypto.randomUUID() }],
        })),
      removeAIModel: (id) =>
        set((state) => ({
          aiModels: state.aiModels.filter((m) => m.id !== id),
        })),
      updateAIModel: (id, updates) =>
        set((state) => ({
          aiModels: state.aiModels.map((m) => (m.id === id ? { ...m, ...updates } : m)),
        })),
      activateAIModel: (id) => set({ activeModelId: id }),

      setPanelMap: (panelMap) => set({ panelMap }),
      swapPanels: (from, to) =>
        set((state) => {
          const newMap = { ...state.panelMap };
          [newMap[from], newMap[to]] = [newMap[to], newMap[from]];
          return { panelMap: newMap };
        }),

      setCollaborators: (collaborators) => set({ collaborators }),

      injectCode: (filename, code, language) =>
        set({
          pendingCodeInjection: { filename, code, language },
        }),
      clearCodeInjection: () => set({ pendingCodeInjection: null }),
    }),
    {
      name: 'yyc3-storage', // LocalStorage key
      partialize: (state) => ({
        // ── Persist only essential state ──
        theme: state.theme,
        viewMode: state.viewMode,
        terminalHeight: state.terminalHeight,
        aiModels: state.aiModels,
        activeModelId: state.activeModelId,
        panelMap: state.panelMap,
      }),
    }
  )
);
```

## 13. 常见问题与最佳实践

### 13.1 TypeScript 常见问题

#### ❌ 问题：`any` 类型滥用

```typescript
// ❌ 避免
function processData(data: any) {
  return data.value;
}

// ✅ 推荐
interface DataItem {
  value: string;
}
function processData(data: DataItem) {
  return data.value;
}
```

#### ❌ 问题：类型断言滥用

```typescript
// ❌ 避免
const data = response as UserData;

// ✅ 推荐
function isUserData(data: unknown): data is UserData {
  return typeof data === 'object' && data !== null && 'id' in data;
}

if (isUserData(response)) {
  // TypeScript 知道 response 是 UserData
}
```

### 13.2 React 常见问题

#### ❌ 问题：依赖项缺失

```typescript
// ❌ 避免
useEffect(() => {
  fetchData(userId);
}, []); // ⚠️ Missing dependency: userId

// ✅ 推荐
useEffect(() => {
  fetchData(userId);
}, [userId]);
```

#### ❌ 问题：状态更新不正确

```typescript
// ❌ 避免：直接修改状态
const [items, setItems] = useState([]);
items.push(newItem); // ❌ 直接修改
setItems(items);

// ✅ 推荐：创建新数组
setItems([...items, newItem]);
```

### 13.3 性能常见问题

#### ❌ 问题：未优化的渲染

```typescript
// ❌ 避免：每次渲染都创建新函数
export function MyComponent() {
  return (
    <ChildComponent onClick={() => doSomething()} />
  );
}

// ✅ 推荐：使用 useCallback
export function MyComponent() {
  const handleClick = useCallback(() => {
    doSomething();
  }, []);

  return <ChildComponent onClick={handleClick} />;
}
```
