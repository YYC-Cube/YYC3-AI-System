## 2. 前端安全防护

### 2.1 XSS 防御

#### 2.1.1 React JSX 自动转义

```typescript

<span>{userInput}</span>  // 自动转义 HTML 特殊字符

// ❌ 危险：绕过自动转义
<div dangerouslySetInnerHTML={{ __html: userInput }} />  // 禁止使用
```

#### 2.1.2 Markdown 渲染安全

```typescript
// /src/app/components/ChatInterface.tsx
// react-markdown 默认安全：不渲染原始 HTML

import ReactMarkdown from 'react-markdown'
import remarkGfm from 'remark-gfm'

// ✅ 安全：react-markdown 默认不解析 HTML 标签
<ReactMarkdown remarkPlugins={[remarkGfm]}>
  {message.content}
</ReactMarkdown>

// 自定义 components 覆盖确保安全渲染
const markdownComponents = {
  // 链接：强制 target="_blank" + rel="noopener noreferrer"
  a: ({ href, children }) => (
    <a href={href} target="_blank" rel="noopener noreferrer">
      {children}
    </a>
  ),
  // 代码块：通过 react-syntax-highlighter 安全渲染
  code: ({ className, children }) => {
    // 语法高亮渲染，不执行代码
    return <SyntaxHighlighter>{children}</SyntaxHighlighter>
  },
}
```

#### 2.1.3 用户输入过滤

```typescript
/**
 * @function sanitizeInput
 * @description 过滤用户输入中的潜在危险字符
 */
function sanitizeInput(input: string): string {
  return input
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#x27;')
    .replace(/\//g, '&#x2F;');
}
```

### 2.3 代码注入安全

#### 2.3.1 注入前验证

```typescript
// 文件名白名单验证
const ALLOWED_EXTENSIONS = /\.(tsx?|jsx?|css|json|md|html|svg)$/;
const MAX_INJECTION_SIZE = 100 * 1024; // 100KB

function validateInjection(filename: string, code: string): boolean {
  // 1. 文件名格式验证
  if (!ALLOWED_EXTENSIONS.test(filename)) {
    console.warn(`Rejected injection: invalid extension for ${filename}`);
    return false;
  }

  // 2. 路径遍历防护
  if (filename.includes('..') || filename.includes('~')) {
    console.warn(`Rejected injection: path traversal attempt in ${filename}`);
    return false;
  }

  // 3. 大小限制
  if (new Blob([code]).size > MAX_INJECTION_SIZE) {
    console.warn(`Rejected injection: code exceeds ${MAX_INJECTION_SIZE} bytes`);
    return false;
  }

  return true;
}
```

### 2.4 Content Security Policy (CSP)

#### 2.4.1 推荐 CSP 配置

```html
<!-- 生产环境 index.html -->
<meta
  http-equiv="Content-Security-Policy"
  content="
  default-src 'self';
  script-src 'self' 'unsafe-eval' blob:;
  style-src 'self' 'unsafe-inline' https://fonts.googleapis.com;
  font-src 'self' https://fonts.gstatic.com;
  img-src 'self' data: blob: https://images.unsplash.com;
  connect-src 'self' https://api.openai.com wss://*.yjs.dev;
  worker-src 'self' blob:;
  frame-src 'none';
"
/>
```

#### 2.4.2 CSP 指令说明

| 指令          | 值                           | 原因                                             |
| ------------- | ---------------------------- | ------------------------------------------------ |
| `script-src`  | `'self' 'unsafe-eval' blob:` | Monaco Editor 需要 `unsafe-eval` 用于 Web Worker |
| `style-src`   | `'self' 'unsafe-inline'`     | Tailwind CSS 运行时注入 + 主题 CSS 变量          |
| `connect-src` | OpenAI + yjs WebSocket       | AI API 调用 + 协同编辑 WebSocket                 |
| `worker-src`  | `blob:`                      | Monaco Editor Web Worker                         |
| `frame-src`   | `'none'`                     | 禁止嵌入 iframe（防止 Clickjacking）             |

```typescript
    const collabs = users.filter(u => u.id !== 'local-user').map(...)
    setCollaborators(collabs)
  }, 2000)

  // 方案 2: 事件驱动（更优）
  const handler = () => {
    const users = collabManager.getUsers()
    const collabs = users.filter(u => u.id !== 'local-user').map(...)
    setCollaborators(collabs)
  }
  collabManager.awareness.on('change', handler)

  return () => {
    window.clearInterval(syncInterval)
    collabManager.awareness.off('change', handler)
  }
}, [setCollaborators])
```

### 9.1 性能标记 API

```typescript
/**
 * @function measurePerformance
 * @description 使用 Performance API 测量关键操作耗时
 */
function measurePerformance(label: string, fn: () => void) {
  performance.mark(`${label}-start`);
  fn();
  performance.mark(`${label}-end`);
  performance.measure(label, `${label}-start`, `${label}-end`);

  const measure = performance.getEntriesByName(label)[0];
  console.debug(`[Perf] ${label}: ${measure.duration.toFixed(2)}ms`);
}

// 使用示例
measurePerformance('theme-switch', () => {
  setTheme(nextTheme(theme));
});
```
