## 4. CI/CD 管道

### 4.1 GitHub Actions 配置

```yaml
# .github/workflows/deploy.yml

name: YYC3 PortAISys CI/CD

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  # ── 代码质量检查 ──
  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
        with:
          version: 9
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: 'pnpm'
      - run: pnpm install --frozen-lockfile
      - run: pnpm tsc --noEmit # TypeScript 类型检查
      - run: pnpm lint # ESLint 检查

  # ── 单元测试 ──
  test:
    runs-on: ubuntu-latest
    needs: lint
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
        with:
          version: 9
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: 'pnpm'
      - run: pnpm install --frozen-lockfile
      - run: pnpm test # Vitest 单元测试

  # ── 构建 ──
  build:
    runs-on: ubuntu-latest
    needs: [lint, test]
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
        with:
          version: 9
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: 'pnpm'
      - run: pnpm install --frozen-lockfile
      - run: pnpm build
      - uses: actions/upload-artifact@v4
        with:
          name: dist
          path: dist/

  # ── 部署到 Vercel ──
  deploy:
    runs-on: ubuntu-latest
    needs: build
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/download-artifact@v4
        with:
          name: dist
      - uses: amondnet/vercel-action@v25
        with:
          vercel-token: ${{ secrets.VERCEL_TOKEN }}
          vercel-org-id: ${{ secrets.VERCEL_ORG_ID }}
          vercel-project-id: ${{ secrets.VERCEL_PROJECT_ID }}
          working-directory: ./dist
```

## 5. 容器化部署

### 5.1 Docker 配置

```dockerfile
# Dockerfile

# ── Stage 1: 构建阶段 ──
FROM node:20-alpine AS builder

# 安装 pnpm
RUN corepack enable && corepack prepare pnpm@9 --activate

WORKDIR /app

# 复制依赖文件
COPY package.json pnpm-lock.yaml ./

# 安装依赖
RUN pnpm install --frozen-lockfile

# 复制源代码
COPY . .

# 构建
RUN pnpm build

# ── Stage 2: 运行阶段 ──
FROM nginx:alpine AS runtime

# 复制 Nginx 配置
COPY nginx.conf /etc/nginx/conf.d/default.conf

# 复制构建产物
COPY --from=builder /app/dist /usr/share/nginx/html

# 健康检查
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s \
  CMD wget -q --spider http://localhost:80/ || exit 1

EXPOSE 80

CMD ["nginx", "-g", "daemon off;"]
```

### 5.2 Nginx 配置

```nginx
# nginx.conf

server {
    listen 80;
    server_name _;
    root /usr/share/nginx/html;
    index index.html;

    # ── SPA 路由支持 ──
    # React Router 使用 BrowserRouter
    # 所有路由回退到 index.html
    location / {
        try_files $uri $uri/ /index.html;
    }

    # ── 静态资源缓存 ──
    # 带 hash 的文件：长期缓存（1 年）
    location /assets/ {
        expires 1y;
        add_header Cache-Control "public, immutable";
        add_header X-Content-Type-Options "nosniff";
    }

    # ── 安全头 ──
    add_header X-Frame-Options "DENY";
    add_header X-Content-Type-Options "nosniff";
    add_header X-XSS-Protection "1; mode=block";
    add_header Referrer-Policy "strict-origin-when-cross-origin";

    # ── Gzip 压缩 ──
    gzip on;
    gzip_types text/plain text/css application/json application/javascript text/xml;
    gzip_min_length 1024;
    gzip_vary on;

    # ── API 代理 ──
    # 代理到后端服务
    location /api/ {
        proxy_pass http://backend:3001/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # ── WebSocket 代理 ──
    # yjs 协同编辑 WebSocket
    location /ws/ {
        proxy_pass http://yjs-server:1234/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_read_timeout 86400;
    }
}
```

| `index.html`      | `no-cache`                    | 每次请求验证（确保最新版本） |

### 7.2 错误监控集成

```typescript
// /src/app/utils/monitoring.ts（推荐实现）

/**
 * @function initErrorMonitoring
 * @description 初始化全局错误监控
 */
function initErrorMonitoring() {
  // 捕获未处理的 JS 错误
  window.addEventListener('error', (event) => {
    reportError({
      type: 'js-error',
      message: event.message,
      filename: event.filename,
      lineno: event.lineno,
      colno: event.colno,
      stack: event.error?.stack,
    });
  });

  // 捕获未处理的 Promise rejection
  window.addEventListener('unhandledrejection', (event) => {
    reportError({
      type: 'promise-rejection',
      reason: String(event.reason),
      stack: event.reason?.stack,
    });
  });
}

/**
 * @function reportError
 * @description 上报错误到监控服务
 */
function reportError(error: Record<string, any>) {
  // 生产环境：发送到 Sentry / DataDog / 自建服务
  if (import.meta.env.VITE_APP_MODE === 'production') {
    fetch('/api/errors', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        ...error,
        timestamp: Date.now(),
        userAgent: navigator.userAgent,
        url: window.location.href,
      }),
    }).catch(() => {}); // 错误上报本身不应影响用户体验
  }

  // 开发环境：控制台输出
  console.error('[YYC3 Error Monitor]', error);
}
```

### 7.3 Performance Observer

```typescript
/**
 * @function initPerformanceMonitoring
 * @description 初始化性能监控
 */
function initPerformanceMonitoring() {
  // 监控 LCP
  const lcpObserver = new PerformanceObserver((list) => {
    const entries = list.getEntries();
    const lastEntry = entries[entries.length - 1];
    console.debug(`[Perf] LCP: ${lastEntry.startTime.toFixed(0)}ms`);
  });
  lcpObserver.observe({ type: 'largest-contentful-paint', buffered: true });

  // 监控 CLS
  let clsValue = 0;
  const clsObserver = new PerformanceObserver((list) => {
    for (const entry of list.getEntries()) {
      if (!(entry as any).hadRecentInput) {
        clsValue += (entry as any).value;
      }
    }
    console.debug(`[Perf] CLS: ${clsValue.toFixed(3)}`);
  });
  clsObserver.observe({ type: 'layout-shift', buffered: true });

  // 监控长任务
  const longTaskObserver = new PerformanceObserver((list) => {
    for (const entry of list.getEntries()) {
      if (entry.duration > 50) {
        console.warn(`[Perf] Long Task: ${entry.duration.toFixed(0)}ms`);
      }
    }
  });
  longTaskObserver.observe({ type: 'longtask' });
}
```

## 9. 回滚策略

### 9.1 快速回滚机制

| 回滚方式                    | 耗时      | 适用场景     |
| --------------------------- | --------- | ------------ |
| **CDN 缓存回退**            | < 1 分钟  | 静态资源问题 |
| **Git Revert + 重新部署**   | 5-10 分钟 | 代码逻辑问题 |
| **Docker 镜像回退**         | 2-5 分钟  | 容器化部署   |
| **Vercel Instant Rollback** | < 30 秒   | Vercel 平台  |

### 9.2 回滚决策树

```
监控告警触发
    ↓
判断严重程度
    ├── Critical (用户无法使用)
    │   → 立即回滚到上一个稳定版本
    │   → 通知团队
    │   → 事后复盘
    │
    ├── Warning (部分功能异常)
    │   → 评估影响范围
    │   → 决定是否回滚或热修复
    │   → 创建紧急修复 PR
    │
    └── Info (非关键问题)
        → 记录 Issue
        → 下个迭代修复
```
