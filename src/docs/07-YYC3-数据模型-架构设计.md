# 07 - YYC3 数据模型 - 架构设计

## 1. 数据模型分层架构

### 1.1 三层数据架构

```
┌────────────────────────────────────────────────────────────┐
│                Layer 3: 持久化层 (Persistent Layer)          │
│  ┌──────────────┐  ┌──────────────┐  ┌──────────────┐     │
│  │ LocalStorage │  │  IndexedDB   │  │  Server DB   │     │
│  │  (Zustand)   │  │ (File Cache) │  │ (Production) │     │
│  └──────────────┘  └──────────────┘  └──────────────┘     │
├────────────────────────────────────────────────────────────┤
│               Layer 2: 状态管理层 (State Layer)              │
│  ┌──────────────┐  ┌──────────────┐  ┌──────────────┐     │
│  │ Zustand Store│  │ Y.Doc (CRDT) │  │ React Query  │     │
│  │  (UI State)  │  │(Collab State)│  │ (Server State│     │
│  └──────────────┘  └──────────────┘  └──────────────┘     │
├────────────────────────────────────────────────────────────┤
│                Layer 1: 数据结构层 (Data Structure)          │
│  ┌──────────────┐  ┌──────────────┐  ┌──────────────┐     │
│  │ Design JSON  │  │ File System  │  │ AI Config    │     │
│  │  (Schema)    │  │  (Virtual)   │  │   (Models)   │     │
│  └──────────────┘  └──────────────┘  └──────────────┘     │
└────────────────────────────────────────────────────────────┘
```

### 1.2 数据流向

```
User Action
    ↓
UI Component
    ↓
Zustand Action
    ↓
State Update (set)
    ↓
├─→ LocalStorage (persist middleware)
│
├─→ Y.Doc (CRDT update)
│   ↓
│   WebSocket Broadcast
│   ↓
│   Remote Clients
│
└─→ UI Re-render (React subscription)
```

### 3.2 状态持久化策略

#### 3.2.1 Persist Middleware 配置

```typescript
// /src/app/store.ts
export const useAppStore = create<AppState>()(
  persist(
    (set) => ({ /* state and actions */ }),
    {
      name: 'yyc3-storage',           // LocalStorage key
      partialize: (state) => ({
        // ── 持久化字段（仅保存必要状态）──
        theme: state.theme,
        viewMode: state.viewMode,
        terminalHeight: state.terminalHeight,
        aiModels: state.aiModels,
        activeModelId: state.activeModelId,
        panelMap: state.panelMap,
        customThemeConfig: state.customThemeConfig,
        
        // ── 不持久化字段（临时状态）──
        // messages: state.messages,           // ❌ 聊天消息不持久化
        // collaborators: state.collaborators, // ❌ 协同状态不持久化
        // terminalVisible: state.terminalVisible, // ❌ 终端显示状态不持久化
      }),
    }
  )
);
```

#### 3.2.2 持久化字段选择原则

| 字段类型 | 是否持久化 | 原因 |
|---------|-----------|------|
| 用户偏好（theme, viewMode） | ✅ | 保持用户体验一致性 |
| AI 模型配置 | ✅ | 避免用户重复配置 |
| 面板布局 | ✅ | 保持工作区布局 |
| 聊天消息 | ❌ | 隐私安全，会话临时性 |
| 协同状态 | ❌ | 实时同步，不需本地持久化 |
| UI 临时状态 | ❌ | 每次打开重置 |

### 3.3 状态更新模式

#### 3.3.1 不可变更新（Immer 风格）

```typescript
// ✅ 推荐：使用展开运算符创建新对象
addMessage: (msg) => set((state) => ({
  messages: [
    ...state.messages,
    { 
      ...msg, 
      id: crypto.randomUUID(), 
      timestamp: Date.now() 
    }
  ]
}));

// ✅ 推荐：使用 map 创建新数组
updateAIModel: (id, updates) => set((state) => ({
  aiModels: state.aiModels.map(m =>
    m.id === id ? { ...m, ...updates } : m
  )
}));

// ❌ 避免：直接修改状态
addMessage: (msg) => set((state) => {
  state.messages.push(msg); // ❌ 直接修改数组
  return state;
});
```

#### 3.3.2 批量更新

```typescript
// ✅ 推荐：单次 set 批量更新
activateAIModel: (id) => set((state) => ({
  aiModels: state.aiModels.map(m => ({
    ...m,
    isActive: m.id === id
  })),
  activeModelId: id
}));

// ❌ 避免：多次 set 分别更新
activateAIModel: (id) => {
  set((state) => ({ aiModels: state.aiModels.map(m => ({ ...m, isActive: m.id === id })) }));
  set({ activeModelId: id }); // ❌ 触发两次重渲染
};
```

```typescript
/**
 * @function syncFileContent
 * @description 同步文件内容到 Y.Doc
 */
export function syncFileContent(filename: string, content: string) {
  const text = yjsDoc.getText(`file:${filename}`);
  
  // 计算差异并应用最小操作
  const currentContent = text.toString();
  if (currentContent !== content) {
    text.delete(0, text.length);
    text.insert(0, content);
  }
}

  const text = yjsDoc.getText(`file:${filename}`);
  return text.toString();
```

#### 4.2.2 SimpleAwareness 实现

```typescript
// /src/app/utils/collaboration.ts

/**
 * @class SimpleAwareness
 * @description 简化的 Awareness 协议实现（生产环境替换为 y-protocols/awareness）
 */
export class SimpleAwareness {
  doc: Y.Doc;
  clientID: number;
  private states: Map<number, CollabAwarenessState> = new Map();
  private listeners: Set<AwarenessChangeHandler> = new Set();

  constructor(doc: Y.Doc) {
    this.doc = doc;
    this.clientID = doc.clientID;
  }

  /**
   * @method getStates
   * @description 获取所有用户状态
   */
  getStates(): Map<number, CollabAwarenessState> {
    return this.states;
  }

  /**
   * @method getLocalState
   * @description 获取本地用户状态
   */
  getLocalState(): CollabAwarenessState | null {
    return this.states.get(this.clientID) || null;
  }

  /**
   * @method setLocalState
   * @description 设置本地用户状态
   */
  setLocalState(state: CollabAwarenessState) {
    const before = this.states.has(this.clientID);
    this.states.set(this.clientID, state);
    this.emit({
      added: before ? [] : [this.clientID],
      updated: before ? [this.clientID] : [],
      removed: [],
    });
  }

  /**
   * @method setRemoteState
   * @description 设置远程用户状态（模拟 WebSocket 接收）
   */
  setRemoteState(clientId: number, state: CollabAwarenessState) {
    const before = this.states.has(clientId);
    this.states.set(clientId, state);
    this.emit({
      added: before ? [] : [clientId],
      updated: before ? [clientId] : [],
      removed: [],
    });
  }

  /**
   * @method on
   * @description 监听状态变更
   */
  on(_event: string, handler: AwarenessChangeHandler) {
    this.listeners.add(handler);
  }

  /**
   * @method off
   * @description 移除监听器
   */
  off(_event: string, handler: AwarenessChangeHandler) {
    this.listeners.delete(handler);
  }

  private emit(changes: { added: number[]; updated: number[]; removed: number[] }) {
    this.listeners.forEach(fn => fn(changes));
  }

  destroy() {
    this.states.clear();
    this.listeners.clear();
  }
}
```

```typescript
/**
 * @function renameFile
 * @description 重命名文件/文件夹
 */
export function renameFile(node: FileNode, newName: string): void {
  const oldPath = node.path;
  node.name = newName;
  node.path = node.path.replace(/[^/]+$/, newName);
  
  // 递归更新子节点路径
  if (node.children) {
    updateChildrenPath(node.children, oldPath, node.path);
  }
}

/**
 * @function updateFileContent
 * @description 更新文件内容
 */
export function updateFileContent(node: FileNode, content: string): void {
  if (node.type !== 'file') return;
  
  node.content = content;
  if (node.metadata) {
    node.metadata.size = content.length;
    node.metadata.updatedAt = Date.now();
  }
}
```

### 5.3 文件搜索与过滤

```typescript
/**
 * @function searchFiles
 * @description 递归搜索文件（支持文件名和内容搜索）
 */
export function searchFiles(
  root: FileNode,
  query: string,
  searchContent: boolean = false
): FileNode[] {
  const results: FileNode[] = [];
  
  function traverse(node: FileNode) {
    // 文件名匹配
    if (node.name.toLowerCase().includes(query.toLowerCase())) {
      results.push(node);
    }
    
    // 内容匹配（仅文件）
    if (searchContent && node.type === 'file' && node.content) {
      if (node.content.toLowerCase().includes(query.toLowerCase())) {
        results.push(node);
      }
    }
    
    // 递归子节点
    if (node.children) {
      node.children.forEach(traverse);
    }
  }
  
  traverse(root);
  return results;
}

/**
 * @function filterByExtension
 * @description 按文件扩展名过滤
 */
export function filterByExtension(root: FileNode, extensions: string[]): FileNode[] {
  const results: FileNode[] = [];
  
  function traverse(node: FileNode) {
    if (node.type === 'file') {
      const ext = node.name.split('.').pop();
      if (ext && extensions.includes(ext)) {
        results.push(node);
      }
    }
    
    if (node.children) {
      node.children.forEach(traverse);
    }
  }
  
  traverse(root);
  return results;
}
```

```typescript
      const response = await fetch(model.endpoint, {

        headers: {

        },

        }),
        signal: AbortSignal.timeout(5000), // 5 秒超时
      });

      if (response.ok) {
        return { success: true, message: '连接成功' };
      } else {
        const error = await response.json();
        return { success: false, message: error.error?.message || '连接失败' };
      }
    }

    // Ollama 协议测试
    if (model.provider === 'ollama') {
      const response = await fetch(model.endpoint, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          model: model.name,
          messages: [{ role: 'user', content: 'test' }],
          stream: false,
        }),
        signal: AbortSignal.timeout(5000),
      });

      if (response.ok) {
        return { success: true, message: '连接成功' };
      } else {
        return { success: false, message: '模型未找到或服务离线' };
      }
    }

    return { success: false, message: '未知提供商' };
  } catch (error) {
    if (error instanceof Error) {
```

```typescript
/**
 * @function syncToServer
 * @description 离线数据同步到服务器（Online First 模式）
 */
export async function syncToServer(data: DesignRoot): Promise<void> {
  if (!navigator.onLine) {
    // 离线状态：保存到 IndexedDB
    await saveToIndexedDB('design-root', data);
    console.log('[Offline] Data saved locally');
    return;
  }

  try {
    // 在线状态：上传到服务器
    const response = await fetch('/api/design', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data),
    });

    if (response.ok) {
      console.log('[Online] Data synced to server');
      await clearIndexedDB('design-root'); // 清除本地缓存
    }
  } catch (error) {
    // 网络错误：回退到离线模式
    await saveToIndexedDB('design-root', data);
    console.error('[Offline Fallback] Data saved locally', error);
  }
}
```

```typescript
    x: z.number().min(0).max(100),
    y: z.number().min(0).max(100),
    w: z.number().min(0).max(100),
    h: z.number().min(0).max(100),
    minW: z.number().optional(),
    minH: z.number().optional(),
    maxW: z.number().optional(),
    maxH: z.number().optional(),
  }),
  style: z.object({
    background: z.string().optional(),
    border: z.string().optional(),
    borderRadius: z.number().optional(),
    padding: z.number().optional(),
    margin: z.number().optional(),
    shadow: z.string().optional(),
  }),
  children: z.array(z.lazy(() => PanelSpecSchema)).optional(),
  components: z.array(ComponentSpecSchema).optional(),
});

/**
 * @const ComponentSpecSchema
 * @description ComponentSpec 数据验证 Schema
 */
export const ComponentSpecSchema = z.object({
  id: z.string().uuid(),
  type: z.string(),
  props: z.record(z.any()),
  style: z.record(z.any()),
  children: z.array(z.lazy(() => ComponentSpecSchema)).optional(),
});

/**
 * @function validateDesignRoot
 * @description 验证 DesignRoot 数据合法性
 */
export function validateDesignRoot(data: unknown): DesignRoot {
  const schema = z.object({
    version: z.string().regex(/^\d+\.\d+\.\d+$/),
    theme: z.enum(['light', 'dark']),
    tokens: z.string(),
    panels: z.array(PanelSpecSchema),
    components: z.array(ComponentSpecSchema),
    styles: z.object({
      tokens: z.any(),
      theme: z.any(),
      components: z.any(),
    }),
  });

  return schema.parse(data);
}
```

## 12. 总结

### 12.1 核心数据模型总览

| 模型 | 用途 | 存储位置 | 持久化 |
|------|------|---------|--------|
| DesignRoot | 设计文档根节点 | Zustand Store | ✅ LocalStorage |
| PanelSpec | 面板布局配置 | DesignRoot.panels | ✅ LocalStorage |
| ComponentSpec | 组件配置 | DesignRoot.components | ✅ LocalStorage |
| AIModel | AI 模型配置 | Zustand Store | ✅ LocalStorage |
| Message | 聊天消息 | Zustand Store | ❌ 内存 |
| FileNode | 虚拟文件系统 | Y.Doc (CRDT) | ✅ WebSocket |
| CollabUser | 协作用户状态 | Awareness | ❌ 实时广播 |
| CustomThemeConfig | 自定义主题 | Zustand Store | ✅ LocalStorage |

### 12.2 数据流设计原则

1. **单一数据源**: Zustand Store 作为唯一真实来源
2. **不可变更新**: 使用展开运算符创建新对象
3. **选择性持久化**: 仅持久化必要状态
4. **CRDT 协同**: 文件内容使用 yjs 自动合并冲突
5. **类型安全**: TypeScript + Zod 双重验证
