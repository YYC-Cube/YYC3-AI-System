**2. 高度估算算法**:
```typescript
function estimateMessageHeight(content: string, role: string): number {
  const baseHeight = 80
  const lineCount = Math.max(1, Math.ceil(content.length / 80))
  const codeBlockCount = (content.match(/```/g) || []).length / 2
  const codeBlockHeight = codeBlockCount * 100
  return baseHeight + (lineCount * 20) + codeBlockHeight + 20
}
```

**3. 自动滚动逻辑**:

```

**5. 性能监控Hook**:
```typescript
export function useMessageListPerformance(messages: Message[]) {
  const renderedCount = messages.length
  const totalHeight = messages.reduce((sum, msg) => sum + estimateMessageHeight(msg.content, msg.role), 0)
  const avgHeight = renderedCount > 0 ? totalHeight / renderedCount : 0
  return {
    renderedCount,
    totalHeight,
    avgHeight,
    virtualizationEnabled: renderedCount > 50,
  }
}
```
