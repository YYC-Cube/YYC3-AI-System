describe('ChatInterface Component', () => {

    vi.clearAllMocks()

    const messages = [
      { id: '1', role: 'user', content: 'Hello', timestamp: Date.now() },
      { id: '2', role: 'assistant', content: 'Hi there!', timestamp: Date.now() },

    render(<ChatInterface messages={messages} onSend={vi.fn()} />)

    expect(screen.getByText('Hello')).toBeInTheDocument()
    expect(screen.getByText('Hi there!')).toBeInTheDocument()
  })

  it('should handle user input', () => {
    const onSend = vi.fn()
    render(<ChatInterface messages={[]} onSend={onSend} />)

    const input = screen.getByPlaceholderText(/输入/i)
    fireEvent.change(input, { target: { value: 'Test message' } })
    fireEvent.keyDown(input, { key: 'Enter' })

    expect(onSend).toHaveBeenCalledWith('Test message')
  })

  it('should display AI responses', () => {
    const messages = [
      { id: '1', role: 'assistant', content: 'AI response', timestamp: Date.now() },
    ]

    render(<ChatInterface messages={messages} onSend={vi.fn()} />)

    expect(screen.getByText('AI response')).toBeInTheDocument()
  })

  it('should handle streaming responses', async () => {
    const onSend = vi.fn().mockResolvedValue({
      content: 'Streaming response',
      done: false,
    })

    render(<ChatInterface messages={[]} onSend={onSend} />)

    const input = screen.getByPlaceholderText(/输入/i)
    fireEvent.change(input, { target: { value: 'Test' } })
    fireEvent.keyDown(input, { key: 'Enter' })

    await waitFor(() => {
      expect(screen.getByText('Streaming response')).toBeInTheDocument()
    })
  })

    const messages = [
      { id: '1', role: 'assistant', content: '**Bold** and *italic*', timestamp: Date.now() },

    expect(screen.getByText('Bold')).toBeInTheDocument()
    expect(screen.getByText('italic')).toBeInTheDocument()

      {
        id: '1',

    expect(screen.getByText(/console\.log/)).toBeInTheDocument()

describe('CodeEditor Component', () => {
  beforeEach(() => {
    vi.clearAllMocks()
  })

    expect(onChange).toHaveBeenCalledWith('const y = 2;')
  })

    const { rerender } = render(<CodeEditor value="" onChange={vi.fn()} language="javascript" />)
    expect(screen.getByRole('textbox')).toBeInTheDocument()

    rerender(<CodeEditor value="" onChange={vi.fn()} language="python" />)
    expect(screen.getByRole('textbox')).toBeInTheDocument()

    rerender(<CodeEditor value="" onChange={vi.fn()} language="typescript" />)
    expect(screen.getByRole('textbox')).toBeInTheDocument()
  })

  it('should provide syntax highlighting', () => {
    render(<CodeEditor value="const x: number = 1;" onChange={vi.fn()} language="typescript" />)

    const editor = screen.getByRole('textbox')
    // 检查语法高亮是否应用（可能需要检查特定的CSS类或属性）
    expect(editor).toBeInTheDocument()
  })

  it('should provide code completion', async () => {
    const onComplete = vi.fn().mockResolvedValue([
      { label: 'console', type: 'function' },
      { label: 'const', type: 'keyword' },
    ])

    render(
      <CodeEditor value="con" onChange={vi.fn()} language="javascript" onComplete={onComplete} />
    )

    const editor = screen.getByRole('textbox')
    fireEvent.change(editor, { target: { value: 'console' } })

    await waitFor(() => {
      expect(onComplete).toHaveBeenCalledWith('console')
    })
  })

  it('should handle file tabs', () => {
    const files = [
      { name: 'App.tsx', content: 'code1', language: 'typescript' },
      { name: 'utils.ts', content: 'code2', language: 'typescript' },
    ]

    render(<CodeEditor files={files} activeFile="App.tsx" onChange={vi.fn()} />)

    expect(screen.getByText('App.tsx')).toBeInTheDocument()
    expect(screen.getByText('utils.ts')).toBeInTheDocument()
  })

  it('should display file status', () => {
    render(
      <CodeEditor
        files={[
          { name: 'App.tsx', content: 'code', language: 'typescript', status: 'saved' },
        ]}
        activeFile="App.tsx"
        onChange={vi.fn()}
      />
    )

    expect(screen.getByText('App.tsx')).toBeInTheDocument()
    // 检查状态指示器（如绿色圆点表示已保存）
  })

  it('should support undo/redo', () => {
    const onChange = vi.fn()
    render(<CodeEditor value="line1\nline2" onChange={onChange} language="javascript" />)

    const editor = screen.getByRole('textbox')

    // 执行撤销操作
    fireEvent.keyDown(editor, {
      key: 'z',
      ctrlKey: true,
      metaKey: true,
    })

    // 执行重做操作
    fireEvent.keyDown(editor, {
      key: 'y',
      ctrlKey: true,
      metaKey: true,
    })

    // 断言编辑器状态
    expect(editor).toBeInTheDocument()
  })

  it('should handle large files', () => {

    const editor = screen.getByRole('textbox')
    expect(editor).toBeInTheDocument()
    expect(editor).toHaveValue(largeCode)
  })
})

describe('FileManager Component', () => {
  beforeEach(() => {
    vi.clearAllMocks()
  })

      {
        name: 'src',
        type: 'folder',
        expanded: true,

          { name: 'App.tsx', type: 'file', language: 'tsx' },
          { name: 'utils.ts', type: 'file', language: 'ts' },
        ],
      },
    ]

    render(<FileManager files={files} onSelect={vi.fn()} />)

    expect(screen.getByText('src')).toBeInTheDocument()
    expect(screen.getByText('App.tsx')).toBeInTheDocument()
    expect(screen.getByText('utils.ts')).toBeInTheDocument()
  })

  it('should handle file selection', () => {
    const onSelect = vi.fn()
    const files = [
      { name: 'App.tsx', type: 'file', language: 'tsx' },
    ]

    render(<FileManager files={files} onSelect={onSelect} />)

    const fileItem = screen.getByText('App.tsx')
    fireEvent.click(fileItem)

    expect(onSelect).toHaveBeenCalledWith('App.tsx')
  })

  it('should expand/collapse directories', () => {
    const files = [
      {
        name: 'src',
        type: 'folder',
        expanded: false,
        children: [{ name: 'App.tsx', type: 'file', language: 'tsx' }],
      },
    ]

    render(<FileManager files={files} onSelect={vi.fn()} />)

    const folder = screen.getByText('src')
    expect(screen.queryByText('App.tsx')).not.toBeInTheDocument()

    fireEvent.click(folder)

    expect(screen.getByText('App.tsx')).toBeInTheDocument()
  })

  it('should display file icons', () => {
    const files = [
      { name: 'App.tsx', type: 'file', language: 'tsx' },

    // 检查文件图标是否存在（可能通过SVG或图片元素）

      img.getAttribute('src')?.includes('tsx')
    )
    const cssIcon = screen.getAllByRole('img').find((img) =>
      img.getAttribute('src')?.includes('css')
    )

    expect(tsxIcon).toBeInTheDocument()
    expect(cssIcon).toBeInTheDocument()
  })

  it('should support file search', () => {
    const files = [
      { name: 'App.tsx', type: 'file', language: 'tsx' },
      { name: 'utils.ts', type: 'file', language: 'ts' },

    const searchInput = screen.getByPlaceholderText(/搜索/i)
    fireEvent.change(searchInput, { target: { value: 'App' } })

    expect(screen.getByText('App.tsx')).toBeInTheDocument()
    expect(screen.queryByText('utils.ts')).not.toBeInTheDocument()
  })

  it('should create new files', () => {
    const onCreateFile = vi.fn()
    render(<FileManager files={[]} onSelect={vi.fn()} onCreateFile={onCreateFile} />)

    const createButton = screen.getByTitle(/新建文件/i)

    expect(onCreateFolder).toHaveBeenCalled()
